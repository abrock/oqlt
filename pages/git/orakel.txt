SERVER-SETUP

1. GRUNDEINSTELLUNGEN

$ cat /etc/conf.d/git-daemon
GITDAEMON_OPTS="--syslog --base-path=/mnt/unit0/git /mnt/unit0/git"
GIT_USER="git"
GIT_GROUP="git"

$ mkdir /mnt/unit0/git
$ chown git:git /mnt/unit0/git
$ useradd -d /mnt/unit0/git git
$ /etc/init.d/git-daemon start
$ rc-update add git-daemon default

2. REPOS EINRICHTEN (1x intern & 1x public)

$ su - git
	$ mkdir /mnt/unit0/git/oqlt_intern.git
	$ chmod 770 /mnt/unit0/git/oqlt_intern.git
	$ cd /mnt/unit0/git/oqlt_intern.git
	$ git init
	$ mkdir /mnt/unit0/git/oqlt.git
	$ chmod 775 /mnt/unit0/git/oqlt.git
	$ cd /mnt/unit0/git/oqlt.git
	$ touch .git/git-daemon-export-ok
	$ git init
